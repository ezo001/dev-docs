#!/usr/bin/env python3
"""
Convert Word documents to Markdown with proper cleanup for Docusaurus/MDX compatibility
Simple regex-based table conversion
"""

import os
import subprocess
import sys
import re
from pathlib import Path

# Configuration
DOCS_DIR = "docs"
INPUT_DIR = "."
OUTPUT_DIR = DOCS_DIR

def clear_old_markdown():
    """Remove old markdown files to ensure clean conversion"""
    print("Clearing old markdown files...")
    if os.path.exists(OUTPUT_DIR):
        for file in Path(OUTPUT_DIR).glob("*.md"):
            if file.name != "intro.md":
                file.unlink()
                print(f"  Removed: {file.name}")

def extract_table_cells(row_html):
    """Extract cell contents from a table row"""
    # Remove the row tags
    row_html = re.sub(r'</?tr[^>]*>', '', row_html)
    # Find all th or td tags and extract their content
    cells = []
    for match in re.finditer(r'<t[hd][^>]*>(.*?)</t[hd]>', row_html, re.DOTALL):
        cell_content = match.group(1).strip()
        # Remove any nested tags but keep the text
        cell_content = re.sub(r'<[^>]+>', '', cell_content)
        # Replace newlines with spaces
        cell_content = cell_content.replace('\n', ' ').strip()
        cells.append(cell_content)
    return cells

def convert_html_tables_to_markdown(content):
    """Convert HTML tables to markdown pipe tables using regex"""
    
    def replace_table(match):
        table_html = match.group(0)
        
        # Extract all rows
        rows = re.findall(r'<tr[^>]*>.*?</tr>', table_html, re.DOTALL)
        
        if not rows:
            return ''
        
        markdown_lines = []
        
        # Find header rows (in thead or first row with th tags)
        thead_match = re.search(r'<thead[^>]*>(.*?)</thead>', table_html, re.DOTALL)
        header_processed = False
        
        for i, row in enumerate(rows):
            cells = extract_table_cells(row)
            
            if not cells:
                continue
            
            # Check if this is a header row
            is_header = (thead_match and row in thead_match.group(0)) or ('<th' in row and not header_processed)
            
            # Create the markdown row
            markdown_row = '| ' + ' | '.join(cells) + ' |'
            markdown_lines.append(markdown_row)
            
            # Add separator after header
            if is_header and not header_processed:
                separator = '| ' + ' | '.join(['---'] * len(cells)) + ' |'
                markdown_lines.append(separator)
                header_processed = True
        
        return '\n\n' + '\n'.join(markdown_lines) + '\n\n'
    
    # Replace all tables
    content = re.sub(r'<table[^>]*>.*?</table>', replace_table, content, flags=re.DOTALL)
    
    return content

def convert_grid_tables_to_pipe(content):
    """Convert Pandoc grid tables to pipe tables"""
    lines = content.split('\n')
    result = []
    i = 0
    
    while i < len(lines):
        line = lines[i]
        
        # Check if this is the start of a grid table (line with +---+)
        if re.match(r'^\s*\+[-=]+\+', line):
            # This is a grid table, collect all lines until the table ends
            table_lines = []
            while i < len(lines) and (lines[i].strip().startswith('+') or lines[i].strip().startswith('|')):
                table_lines.append(lines[i])
                i += 1
            
            # Parse the grid table
            pipe_table = parse_grid_table(table_lines)
            result.extend(pipe_table)
            continue
        
        result.append(line)
        i += 1
    
    return '\n'.join(result)

def parse_grid_table(table_lines):
    """Parse a grid table and convert to pipe table"""
    # Extract rows (lines that start with |, not +)
    data_rows = [line for line in table_lines if line.strip().startswith('|') and not line.strip().startswith('|=')]
    
    if not data_rows:
        return []
    
    pipe_rows = []
    for row in data_rows:
        # Split by | and clean up
        cells = [cell.strip() for cell in row.split('|')[1:-1]]  # Remove first and last empty elements
        
        # Remove HTML tags from each cell
        cleaned_cells = []
        for cell in cells:
            # Remove all HTML tags
            cell = re.sub(r'<[^>]+>', '', cell)
            cleaned_cells.append(cell)
        
        if cleaned_cells and any(cell for cell in cleaned_cells):  # Only add non-empty rows
            pipe_rows.append('| ' + ' | '.join(cleaned_cells) + ' |')
    
    # Add separator after first row (header)
    if len(pipe_rows) > 1:
        num_cols = pipe_rows[0].count('|') - 1
        separator = '| ' + ' | '.join(['---'] * num_cols) + ' |'
        pipe_rows.insert(1, separator)
    
    return [''] + pipe_rows + ['']

def clean_markdown_content(content):
    """Clean up markdown content to be MDX-compatible"""
    
    # Convert grid tables to pipe tables FIRST
    content = convert_grid_tables_to_pipe(content)
    
    # Convert HTML tables to markdown (in case there are any)
    content = convert_html_tables_to_markdown(content)
    
    # NOW remove HTML tags (after table conversion)
    # Remove nested HTML tags within markdown bold (e.g., **<strong>text</strong>** -> **text**)
    content = re.sub(r'\*\*<strong>(.*?)</strong>\*\*', r'**\1**', content)
    content = re.sub(r'\*\*<b>(.*?)</b>\*\*', r'**\1**', content)
    
    # Remove ALL remaining HTML formatting tags
    content = re.sub(r'</?strong>', '', content)
    content = re.sub(r'</?b>', '', content)
    content = re.sub(r'</?em>', '', content)
    content = re.sub(r'</?i>', '', content)
    content = re.sub(r'</?u>', '', content)
    content = re.sub(r'</?span[^>]*>', '', content)
    content = re.sub(r'</?colgroup[^>]*>', '', content)
    content = re.sub(r'<col[^>]*/?>', '', content)
    
    # Remove Pandoc-style formatting attributes
    content = re.sub(r'\[([^\]]+)\]\{[^}]+\}(\([^)]+\))', r'[\1]\2', content)
    
    # Remove image width/height attributes
    content = re.sub(r'(\!\[[^\]]*\]\([^)]+\))\{[^}]+\}', r'\1', content)
    content = re.sub(r'\{[.#][^}]+\}', '', content)
    
    # Fix image paths
    content = re.sub(r'!\[([^\]]*)\]\(docs/media/', r'![\1](media/', content)
    
    # Convert HTML img tags to markdown
    def img_to_markdown(match):
        img_tag = match.group(0)
        src_match = re.search(r'src="([^"]*)"', img_tag)
        alt_match = re.search(r'alt="([^"]*)"', img_tag)
        if src_match:
            src = src_match.group(1).replace('docs/media/', 'media/')
            alt = alt_match.group(1) if alt_match else ''
            return f'![{alt}]({src})'
        return ''
    
    content = re.sub(r'<img[^>]*/?>', img_to_markdown, content)
    
    return content

def convert_docx_to_markdown(docx_file):
    """Convert a single DOCX file to Markdown using pandoc"""
    try:
        input_path = Path(INPUT_DIR) / docx_file
        output_filename = docx_file.replace('.docx', '.md')
        output_path = Path(OUTPUT_DIR) / output_filename
        
        media_dir = Path(OUTPUT_DIR) / 'media'
        media_dir.mkdir(exist_ok=True)
        
        print(f"Converting: {docx_file}")
        
        # Use plain markdown which creates grid tables (we'll convert them)
        result = subprocess.run(
            [
                'pandoc',
                str(input_path),
                '-f', 'docx',
                '-t', 'markdown',  # Plain markdown with grid tables
                '--wrap=none',
                '--extract-media', str(OUTPUT_DIR),
                '-o', str(output_path)
            ],
            capture_output=True,
            text=True,
            check=True
        )
        
        # Read the generated markdown
        with open(output_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # Clean and convert tables
        cleaned_content = clean_markdown_content(content)
        
        # Add frontmatter
        if not cleaned_content.startswith('---'):
            doc_title = output_filename.replace('.md', '').replace('_', ' ')
            frontmatter = f"""---
sidebar_position: 2
title: {doc_title}
---

"""
            cleaned_content = frontmatter + cleaned_content
        
        # Write back
        with open(output_path, 'w', encoding='utf-8') as f:
            f.write(cleaned_content)
        
        print(f"  ✓ Converted to {output_filename}")
        
        if media_dir.exists() and any(media_dir.iterdir()):
            print(f"  ✓ Extracted images to {media_dir}")
        
        return True
        
    except subprocess.CalledProcessError as e:
        print(f"  ✗ Error converting {docx_file}")
        print(f"    {e.stderr}")
        return False
    except Exception as e:
        print(f"  ✗ Unexpected error with {docx_file}: {e}")
        return False

def main():
    """Main conversion process"""
    print("=" * 70)
    print("CONVERTING WORD DOCUMENTS TO MARKDOWN")
    print("=" * 70)
    
    os.makedirs(OUTPUT_DIR, exist_ok=True)
    clear_old_markdown()
    
    docx_files = [f for f in os.listdir(INPUT_DIR) 
                  if f.endswith('.docx') and not f.startswith('~$')]
    
    if not docx_files:
        print("No .docx files found in current directory")
        return 1
    
    print(f"Found {len(docx_files)} documents to convert")
    
    success_count = 0
    for docx_file in docx_files:
        if convert_docx_to_markdown(docx_file):
            success_count += 1
    
    print("=" * 70)
    print("CONVERSION COMPLETE")
    print(f"  Success: {success_count}/{len(docx_files)}")
    print(f"  Failed: {len(docx_files) - success_count}/{len(docx_files)}")
    print("=" * 70)
    
    if success_count == len(docx_files):
        print("Next steps:")
        print("1. Run: npm start")
        print("2. Check tables are rendering correctly")
        print("3. If good: git add . && git commit && git push")
    
    print("=" * 70)
    return 0

if __name__ == "__main__":
    sys.exit(main())
